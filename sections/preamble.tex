%% LianTze 7 Dec 2016:
%% Updated how the wordcount is implemented
\newcommand\wordcount{%
  \immediate\write18{texcount -utf8 -merge -sum -incbib -dir -sub=none -brief \jobname.tex | cut -d : -f 1 > 'count.txt'}%
  \input{count.txt}\ignorespaces words%
}
